\begin{frame}
	\frametitle{Профиль $r$-связности}
	\begin{itemize}
		\item Задача бинарной классификации: $Y = \{1, -1\}$
		\item $S_2 = \{e, h\} \colon Y \rightarrow Y$
		\item Группа $G = (S_2)^L$, элемент $g \in G = \{h_1, h_2, \dots, h_L\}$
\[
    (g a)(x_i) =
    \begin{cases}
        a(x_i), \text { при } h_i = e \\
        1-a(x_i), \text { при } h_i \neq e.& \\
    \end{cases}
\]
		\item Действие G "--- изометрия (по расстоянию Хэмминга)
		\item Профиль $r$-связности $A$ "--- функцию от параметра $q$:
\[
    \Theta_r(q, A) = \sum_{a \in A} [|B_r(a, A)| = q].
\]
Значение $\Theta_r(q, A)$ соответствует числу алгоритмов $a \in A$, имеющих ровно $q$ соседей в окрестности $B_r(a, A)$.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Профиль расслоения-связности}
	\begin{itemize}
		\item Профиль расслоения $\Delta(m, A)$:
\[
    \Delta(m, A) = \sum_{a \in A} [ n(a, \XX) = m].
\]
		\item Профиль расслоения-связности $\Lambda_r(m, q, A)$ семейства $A$:
\[
    \Lambda_r(m, q, A) = \Big|\{a \in A \colon |B_r(a, A)| = q\text{ и } n(a, \XX) = m \}\Big|.
\]
	(число алгоритмов $a \in A$ с $m$ ошибками 
	и $q$ соседями в окрестности радиуса $r$)
	
	\item Теорема о наиболее вероятном профиле расслоения:
\[
    \frac {1}{|G|} \sum_{g \in G} \Delta(m, g A) = \frac{C_L^m}{2^L} |A|.
\]
	\item  Теорема о профиле расслоения-связности
\[
    \frac {1}{|G|} \sum_{g \in G} \Lambda_r(m, q, g A) = \frac{C_L^m}{2^L} \cdot \Theta_r(q, A)
\]	

\end{itemize}
	
\end{frame}

\begin{itemize}
        \item Пусть $A_m^n$ "--- множество из $n$ алгоритмов, каждый из которых допускает $m$ ошибок на полной выборке. Векторы ошибок независимы.
\end{itemize}

\begin{Th}[Вероятность переобучения для $A_m^n$]
    Пусть $\mu$ "--- рандомизированный МЭР. Тогда
    \[\E_G Q_{\mu}(\epsilon, A_m^n) = 1 - \left(1 - Q_{\mu}(\epsilon, a_m) \right)^n\]
\end{Th}

\end{frame}

\begin{frame}
\frametitle{Отсутствие связности: случайные множества алгоритмов}
\begin{itemize}
    \item Пронумеруем алгоритмы: $A = (a_1, \dots, a_d)$
    \item $S_L$ "--- симметрическая группа порядка $L$;
    \begin{itemize}
%        \item $|S_L| = L!$
        \item $S_L$ действует на объектах выборки;
        \item $S_L$ действует на векторах ошибок алгоритмов;
    \end{itemize}
    \item $G = (S_L)^{d}$ "--- свободное произведение $S_L$;
    \begin{itemize}
    \item $g = (g_1, \dots, g_d) \in G$ "--- элемент группы $G$, $g_i \in S_L$;
        \item $G$ действует на векторе алгоритмов: \[gA = \Big(g_1(a_1), \dots, g_d(a_d)\Big).\]
    \end{itemize}
    \item Вероятность переобучения несвязной перестановки $A$:
        \[\bar{Q}_\mu(\epsilon, A) = \E_G  Q_\mu(\epsilon, gA),  \text{ где } \E_G \stackrel{def}{=} \frac{1}{(L!)^d} \sum_{g \in G} \]
    \item $\bar{Q}_\mu(\epsilon, A)$ зависит только от профиля расслоения $A$.
\end{itemize}

\end{frame}
